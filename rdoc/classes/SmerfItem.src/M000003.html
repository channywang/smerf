    def validate_field(hash, field, msg)
      error = ""
      if (hash.blank?() or !hash.has_key?(field) or hash[field].blank?)        
        error = ((msg.size > 0) ? msg : "No '#{field}' specified for ") + object_id_message() + "\n"
      end
      return error
    end
