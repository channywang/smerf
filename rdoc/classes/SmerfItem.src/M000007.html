    def decode_data()
      # Raw data contains two array items, 
      # [0] = name 
      # [1] = data
      if @raw_data.blank? or @raw_data.size < 2
        raise("Invalid data found in form")
      end
      @item_tag = @raw_data[0]
      @item_data = @raw_data[1]
    end
