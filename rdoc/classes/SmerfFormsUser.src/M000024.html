  def SmerfFormsUser.find_user_smerf_form(user_id, smerf_form_id)
    # Find the form record for the current user
    smerf_forms_user = nil
    if (user_id > 0)
      smerf_forms_users = SmerfFormsUser.find(:first, 
        :conditions => ['smerf_form_id = ? AND user_id = ?', 
        smerf_form_id, user_id])
    else
      raise(RuntimeError, 
        "For the form responses to be saved for a user, a record ID for 
        the user table needs to be specified. This can be set by using 
        setter function SmerfForm.smerf_user_id, e.g. SmerfForm.smerf_user_id = 1")
    end 
    
    return smerf_forms_users
  end
