  def cache
    # Unserialize the data
    cache = unserialize_attribute('cache')
    # Initialize class variables
    cache[:smerfform].init_class_variables() if (cache and 
      cache[:smerfform] and cache[:smerfform].object_index.length() <= 0)
    self[:cache] = cache
  end
